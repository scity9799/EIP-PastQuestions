정처기 C언어 포인터

1. 포인터(pointer)
   메모리 주소를 저장하는 변수
   일반 변수 -> 값 저장
   포인터 변수 -> 주소 저장

   int num = 10;
   int *p = &num;
   
   num : 값 10 저장
   &num : num의 주소
   p : num의 주소를 저장하는 포인터

   & : 주소연산자
   * : 간접 참조 연산자

   *p = 20;

2) 포인터와 배열
   int arr[3] = {1, 2, 3};
   int *p = arr;

   arr == &arr[0]
   p[1] == *(p + 1)
   p[0] => 첫번째 요소
   
   p[i] == *(p + i)
   p[0] == *(p + 0) == *p
   
   p -> arr[0] (1)
   p : arr[0] 주소
   *p : 그 주소의 값 -> 1
   p[0] : 동일 -> 1

p[0] == *p   항상 같음
arr[0] == *arr    항상 같음
*p + 1      값 + 1
*(p + 1)   다음 요소

   p가 null일 때 p[0]?
   int *p = NULL;
   p[0] => *(p + 0) == *p
   정상적인 값 없음
   Segmentation Fault 런타임 오류

3) 배열과 포인터 차이
   배열   포인터
값변경   불가   가능
주소   고정   변경가능
   arr = p   p = arr

4) 이중 포인터
   int num = 10
   int *p = &num;
   int **pp = &p;

   p -> num의 주소
   pp -> p의 주소

5) 함수와 포인터
   값을 직접 바꾸려면 포인터 사용
   void change(int *x){
      *x = 100;
   }

+) *p++      (*p)++
   int arr[3] = {10, 20, 30};
   int *p = arr;
   int a = *p++;
   int b = (*p)++;
   a, b의 값은?

   *p++    *(p++)
      a => 10
      p => arr[1]

   (*p)++
      현재 p = arr[1]
      b = 20
      arr[1] => 21

   a = 10
   b = 20
   arr = {10, 21, 30}

+) 포인터이동 + 값변경
   int arr[3] = {1, 2, 3};
   int *p = arr;

   *(p + 1) = *(p) + 5;
   arr의 결과?

   *(p+1) => arr[1] = 2
   *(p) => arr[0] = 1
   arr[1] = 1 + 5
   {1, 6, 3}

+) 이중포인터
   int a = 5;
   int *p = &a;
   int **pp = &p;

   **pp = 10;
   a의 값은?

   pp -> p -> a
   **p -> a
   10

+) 연산 우선순위 + 이중 포인터
   int a = 3;
   int *p = &a;
   int **pp = &p;

   int x = **p + 2;

   x의 값은?

   **p -> a -> 3 + 2 = 5

   *p++   : 포인터 이동
   (*p)++  : 값증가
   p[i] == *(p + i)
   **pp   : 2번 따라가서 값


#include <stdio.h>
 
void func(int** arr, int size){
  for(int i=0; i<size; i++){
     *(*arr + i) = (*(*arr+i) + i) % size;
  }
}

   *(*arr + i) 
   arr = pp = &p
   *arr = p
   *arr + i = p + i
   *(*arr + i) = *(p + i) = arr[i]

   *(*arr + i) = (*(*arr + i) + i) % size;
   arr[i] = (arr[i] + i) % 5;
   
   arr = {3, 1, 4, 1, 5}
   i = 0
   arr[0] = (3 + 0) % 5 => 3
   i = 1
   arr[1] = (1 + 1) % 5 => 2
   i = 2   => 1
   i = 3    => 4
   i = 4   => 4
   {3, 2, 1, 4, 4}
 
int main(){
  int arr[] = {3,1, 4, 1, 5};
  int* p = arr;
  int** pp = &p;         pp -> p -> arr[0]
  int num = 6;
  
  func(pp, 5);  
  num = arr[2];
  printf("%d", num);           arr = 1

}
